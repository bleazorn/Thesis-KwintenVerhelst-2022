p	::=	(module info tail)

info	::= 	(#:from-contract (info/c (locals (aloc ...)) (conflicts ((aloc (aloc ...)) ...))))

tail  	::= 	(begin effect ... tail)
	|	(halt triv)
 	 	 	 	 
effect	::=	(set! aloc triv)
 	|	(set! aloc (binop aloc aloc))
	|	(begin effect ... effect)
 	 	 
triv	::= 	integer
	|	aloc
	 	 
aloc	::= 	aloc?
 	 		 	 
binop	::=	+
 	|	*
 	 	 	 	 
integer	::=	integer32?

undead-analysis

p	::=	(module info tail)

info	::= 	(#:from-contract (info/c (locals (aloc ...))))

tail  	::= 	(begin effect ... tail)
	|	(halt triv)
 	 	 	 	 
effect	::=	(set! aloc triv)
 	|	(set! aloc (binop aloc aloc))
	|	(set! aloc (+ aloc int12))
	|	(begin effect ... effect)
 	 	 
triv	::= 	integer
	|	aloc
	 	 
aloc	::= 	aloc?
 	 		 	 
binop	::=	+
 	|	*
 	 	 	 	 
integer	::=	int32
	|	int12

int32	::=	int32?
int12	::=	int12?
